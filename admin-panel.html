🔐 Diagnostic & Correction RLS Supabase

Ce panel utilise la clé anon sans session Supabase Auth. Pour que l'admin puisse lire/écrire les tables, des Row Level Security policies permissives doivent être configurées. Copiez et exécutez le SQL ci-dessous dans Supabase → SQL Editor. 

-- ═══════════════════════════════════════════════
-- RENAUD EXPRESS — Fix RLS pour admin (clé anon)
-- Exécuter dans Supabase → SQL Editor
-- ═══════════════════════════════════════════════

-- 1. TABLE CLIENTS (lecture admin)
DROP POLICY IF EXISTS "admin_read_clients" ON clients;
CREATE POLICY "admin_read_clients"
  ON clients FOR SELECT TO anon
  USING (true);

-- 2. TABLE DELIVERY_ZONES (lecture publique + écriture admin)
DROP POLICY IF EXISTS "public_read_delivery_zones" ON delivery_zones;
CREATE POLICY "public_read_delivery_zones"
  ON delivery_zones FOR SELECT TO anon
  USING (true);

DROP POLICY IF EXISTS "admin_insert_delivery_zones" ON delivery_zones;
CREATE POLICY "admin_insert_delivery_zones"
  ON delivery_zones FOR INSERT TO anon
  WITH CHECK (true);

DROP POLICY IF EXISTS "admin_update_delivery_zones" ON delivery_zones;
CREATE POLICY "admin_update_delivery_zones"
  ON delivery_zones FOR UPDATE TO anon
  USING (true) WITH CHECK (true);

DROP POLICY IF EXISTS "admin_delete_delivery_zones" ON delivery_zones;
CREATE POLICY "admin_delete_delivery_zones"
  ON delivery_zones FOR DELETE TO anon
  USING (true);

-- 3. TABLE SERVICES (écriture admin)
DROP POLICY IF EXISTS "admin_update_services" ON services;
CREATE POLICY "admin_update_services"
  ON services FOR UPDATE TO anon
  USING (true) WITH CHECK (true);
